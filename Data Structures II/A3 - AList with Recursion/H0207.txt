void resize()
{
    int newSize;
    Entry * newTable;

    newSize = nextPrime(2 * capacity);
    newTable = new Entry[newSize];

    for (int i = 0; i < capacity; i++)
    {
        if (table[i].isOccupied && !table[i].isRemoved)
        {
            int loc = table[i].data % newSize;

            while (newTable[loc].isOccupied)
                loc = (loc + 1) % newSize;

            newTable[loc] = table[i];
        }
    }

    delete[] table;
    table = newTable;
    capacity = newSize;
}
